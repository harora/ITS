72 int NicEntryDebug::collectGates(const char* pattern, GateStack& gates)
73 {
74  cModule* host = nicPtr->getParentModule();
75  int i = 1;
76  char gateName[20];
77  //create the unique name for the gate (composed of the nic module id and a counter)
78  sprintf(gateName, pattern, nicId, i);
79  while(host->hasGate(gateName))
80  {
81  cGate* hostGate = host->gate(gateName);
82  if(hostGate->isConnectedOutside()) {
83  opp_error("Gate %s is still connected but not registered with this "
84  "NicEntry. Either the last NicEntry for this NIC did not "
85  "clean up correctly or another gate creation module is "
86  "interfering with this one!", gateName);
87  }
88  assert(hostGate->isConnectedInside());
89  gates.push_back(hostGate);
90 
91  ++i;
92  sprintf(gateName, pattern, nicId, i);
93  }
94 
95  return i - 1;
96 }

114 cGate* NicEntryDebug::requestInGate(void) {
115  collectFreeGates();
116 
117  // gate of the host
118  cGate *hostGate;
119 
120  if (!freeInGates.empty()) {
121  hostGate = freeInGates.back();
122  freeInGates.pop_back();
123  } else {
124  char gateName[20];
125 
126  // we will have one more in gate
127  ++inCnt;
128 
129  //get a unique name for the gate (composed of the nic module id and a counter)
130  sprintf(gateName, "in%d-%d", nicId, inCnt);
131 
132  // create a new gate for the host module
133  nicPtr->getParentModule()->addGate(gateName, cGate::INPUT);
134  hostGate = nicPtr->getParentModule()->gate(gateName);
135 
136  // gate of the nic
137  cGate *nicGate;
138 
139  // create a new gate for the nic module
140  nicPtr->addGate(gateName, cGate::INPUT);
141  nicGate = nicPtr->gate(gateName);
142 
143  // connect the hist gate with the nic gate
144  hostGate->connectTo(nicGate);
145 
146  // pointer to the phy module
147  ChannelAccess* phyModule;
148  // gate of the phy module
149  cGate *phyGate;
150 
151  // to avoid unnecessary dynamic_casting we check for a "phy"-named submodule first
152  if ((phyModule = dynamic_cast<ChannelAccess *> (nicPtr->getSubmodule("phy"))) == NULL)
153  phyModule = FindModule<ChannelAccess*>::findSubModule(nicPtr);
154  assert(phyModule != 0);
155 
156  // create a new gate for the phy module
157  phyModule->addGate(gateName, cGate::INPUT);
158  phyGate = phyModule->gate(gateName);
159 
160  // connect the nic gate (the gate of the compound module) to
161  // a "real" gate -- the gate of the phy module
162  nicGate->connectTo(phyGate);
163  }
164 
165  return hostGate;
166 }
