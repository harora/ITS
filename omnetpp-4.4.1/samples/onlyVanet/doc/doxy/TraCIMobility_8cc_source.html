140 {

189 void TraCIMobility::changePosition()
190 {
191  // ensure we're not called twice in one time step
192  ASSERT(lastUpdate != simTime());
193 
194  // keep statistics (for current step)
195  currentPosXVec.record(move.getStartPos().x);
196  currentPosYVec.record(move.getStartPos().y);
197 
198  Coord nextPos = calculateAntennaPosition(roadPosition);
199  nextPos.z = move.getCurrentPosition().z;
200 
201  // keep statistics (relative to last step)
202  if (statistics.startTime != simTime()) {
203  simtime_t updateInterval = simTime() - this->lastUpdate;
204 
205  double distance = move.getStartPos().distance(nextPos);
206  statistics.totalDistance += distance;
207  statistics.totalTime += updateInterval;
208  if (speed != -1) {
209  statistics.minSpeed = std::min(statistics.minSpeed, speed);
210  statistics.maxSpeed = std::max(statistics.maxSpeed, speed);
211  currentSpeedVec.record(speed);
212  if (last_speed != -1) {
213  double acceleration = (speed - last_speed) / updateInterval;
214  double co2emission = calculateCO2emission(speed, acceleration);
215  currentAccelerationVec.record(acceleration);
216  currentCO2EmissionVec.record(co2emission);
217  statistics.totalCO2Emission+=co2emission * updateInterval.dbl();
218  }
219  last_speed = speed;
220  } else {
221  last_speed = -1;
222  speed = -1;
223  }
224  }
225  this->lastUpdate = simTime();
226 
227  move.setStart(Coord(nextPos.x, nextPos.y, move.getCurrentPosition().z)); // keep z position
228  move.setDirectionByVector(Coord(cos(angle), -sin(angle)));
229  move.setSpeed(speed);
230  if (ev.isGUI()) updateDisplayString();
231  fixIfHostGetsOutside();
232  updatePosition();
233 }

312 double TraCIMobility::calculateCO2emission(double v, double a) const {
313  // Calculate CO2 emission parameters according to:
314  // Cappiello, A. and Chabini, I. and Nam, E.K. and Lue, A. and Abou Zeid, M., "A statistical model of vehicle emissions and fuel consumption," IEEE 5th International Conference on Intelligent Transportation Systems (IEEE ITSC), pp. 801-809, 2002
315 
316  double A = 1000 * 0.1326; // W/m/s
317  double B = 1000 * 2.7384e-03; // W/(m/s)^2
318  double C = 1000 * 1.0843e-03; // W/(m/s)^3
319  double M = 1325.0; // kg
320 
321  // power in W
322  double P_tract = A*v + B*v*v + C*v*v*v + M*a*v; // for sloped roads: +M*g*sin_theta*v
323 
324  /*
325  // "Category 7 vehicle" (e.g. a '92 Suzuki Swift)
326  double alpha = 1.01;
327  double beta = 0.0162;
328  double delta = 1.90e-06;
329  double zeta = 0.252;
330  double alpha1 = 0.985;
331  */
332 
333  // "Category 9 vehicle" (e.g. a '94 Dodge Spirit)
334  double alpha = 1.11;
335  double beta = 0.0134;
336  double delta = 1.98e-06;
337  double zeta = 0.241;
338  double alpha1 = 0.973;
339 
340  if (P_tract <= 0) return alpha1;
341  return alpha + beta*v*3.6 + delta*v*v*v*(3.6*3.6*3.6) + zeta*a*v;
342 }
