78 void VANETMobilityBase::initialize(int stage)
79 {
80  cSimpleModule::initialize(stage);
81  EV_TRACE << "initializing VANETMobilityBase stage " << stage << endl;
82  if (stage == 0)
83  {
84  constraintAreaMin.x = par("constraintAreaMinX");
85  constraintAreaMin.y = par("constraintAreaMinY");
86  constraintAreaMin.z = par("constraintAreaMinZ");
87  constraintAreaMax.x = par("constraintAreaMaxX");
88  constraintAreaMax.y = par("constraintAreaMaxY");
89  constraintAreaMax.z = par("constraintAreaMaxZ");
90  visualRepresentation = findVisualRepresentation();
91  if (visualRepresentation) {
92  const char *s = visualRepresentation->getDisplayString().getTagArg("p", 2);
93  if (s && *s)
94  error("The coordinates of '%s' are invalid. Please remove automatic arrangement"
95  " (3rd argument of 'p' tag) from '@display' attribute.", visualRepresentation->getFullPath().c_str());
96  }
97  }
98  else if (stage == 1)
99  {
100  initializePosition();
101  if (!isFiniteNumber(lastPosition.x) || !isFiniteNumber(lastPosition.y) || !isFiniteNumber(lastPosition.z))
102  throw cRuntimeError("Mobility position is not a finite number after initialize (x=%g,y=%g,z=%g)", lastPosition.x, lastPosition.y, lastPosition.z);
103  if (isOutside())
104  throw cRuntimeError("Mobility position (x=%g,y=%g,z=%g) is outside the constraint area (%g,%g,%g - %g,%g,%g)",
105  lastPosition.x, lastPosition.y, lastPosition.z,
106  constraintAreaMin.x, constraintAreaMin.y, constraintAreaMin.z,
107  constraintAreaMax.x, constraintAreaMax.y, constraintAreaMax.z);
108  EV << "initial position. x = " << lastPosition.x << " y = " << lastPosition.y << " z = " << lastPosition.z << endl;
109  emitMobilityStateChangedSignal();
110  updateVisualRepresentation();
111 
112  if (!areaInitalized)
113  {
114  mininumArea=constraintAreaMin;
115  maximumArea = constraintAreaMax;
116  updateDisplayString();
117  areaInitalized=true;
118  }
119  else
120  {
121 
122  bool change=false;
123  if (isFiniteNumber(mininumArea.x) && mininumArea.x>constraintAreaMin.x)
124  {
125  mininumArea.x=constraintAreaMin.x;
126  change=true;
127  }
128  if (isFiniteNumber(mininumArea.y) && mininumArea.y>constraintAreaMin.y)
129  {
130  mininumArea.y=constraintAreaMin.y;
131  change=true;
132  }
133  if (isFiniteNumber(mininumArea.z) && mininumArea.z>constraintAreaMin.z)
134  {
135  mininumArea.z=constraintAreaMin.z;
136  change=true;
137  }
138  if (isFiniteNumber(maximumArea.x) && maximumArea.x<constraintAreaMax.x)
139  {
140  maximumArea.x=constraintAreaMax.x;
141  change=true;
142  }
143  if (isFiniteNumber(maximumArea.y) && maximumArea.y<constraintAreaMax.y)
144  {
145  maximumArea.y=constraintAreaMax.y;
146  change=true;
147  }
148  if (isFiniteNumber(maximumArea.z) && maximumArea.z<constraintAreaMax.z)
149  {
150  maximumArea.z=constraintAreaMax.z;
151  change=true;
152  }
153  if (change)
154  updateDisplayString();
155  }
156 
157  }

168 void VANETMobilityBase::setInitialPosition()
169 {
170  // reading the coordinates from omnetpp.ini makes predefined scenarios a lot easier
171  bool filled = false;
172  if (hasPar("initFromDisplayString") && par("initFromDisplayString").boolValue() && visualRepresentation)
173  {
174  filled = parseIntTo(visualRepresentation->getDisplayString().getTagArg("p", 0), lastPosition.x)
175  && parseIntTo(visualRepresentation->getDisplayString().getTagArg("p", 1), lastPosition.y);
176  if (filled)
177  lastPosition.z = 0;
178 
179  }
180  // not all mobility models have "initialX", "initialY" and "initialZ" parameters
181  else if (hasPar("initialX") && hasPar("initialY") && hasPar("initialZ"))
182  {
183  lastPosition.x = par("initialX");
184  lastPosition.y = par("initialY");
185  lastPosition.z = par("initialZ");
186  filled = true;
187  }
188  if (!filled)
189  lastPosition = getRandomPosition();
190 }

242 static int reflect(double min, double max, double &coordinate, double &speed)
243 {
244  double size = max - min;
245  double value = coordinate - min;
246  int sign = 1 - FWMath::modulo(floor(value / size), 2) * 2;
247  ASSERT(sign == 1 || sign == -1);
248  coordinate = FWMath::modulo(sign * value, size) + min;
249  speed = sign * speed;
250  return sign;
251 }

253 void VANETMobilityBase::reflectIfOutside(Coord& targetPosition, Coord& speed, double& angle)
254 {
255  int sign;
256  double dummy;
257  if (lastPosition.x < constraintAreaMin.x || constraintAreaMax.x < lastPosition.x) {
258  sign = reflect(constraintAreaMin.x, constraintAreaMax.x, lastPosition.x, speed.x);
259  reflect(constraintAreaMin.x, constraintAreaMax.x, targetPosition.x, dummy);
260  angle = 90 + sign * (angle - 90);
261  }
262  if (lastPosition.y < constraintAreaMin.y || constraintAreaMax.y < lastPosition.y) {
263  sign = reflect(constraintAreaMin.y, constraintAreaMax.y, lastPosition.y, speed.y);
264  reflect(constraintAreaMin.y, constraintAreaMax.y, targetPosition.y, dummy);
265  angle = sign * angle;
266  }
267  if (lastPosition.z < constraintAreaMin.z || constraintAreaMax.z < lastPosition.z) {
268  sign = reflect(constraintAreaMin.z, constraintAreaMax.z, lastPosition.z, speed.z);
269  reflect(constraintAreaMin.z, constraintAreaMax.z, targetPosition.z, dummy);
270  // NOTE: angle is not affected
271  }
272 }

316 void VANETMobilityBase::handleIfOutside(BorderPolicy policy, Coord& targetPosition, Coord& speed, double& angle)
317 {
