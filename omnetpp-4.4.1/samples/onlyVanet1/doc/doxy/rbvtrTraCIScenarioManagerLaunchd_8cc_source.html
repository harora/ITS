38 void rbvtrTraCIScenarioManagerLaunchd::initialize(int stage)
39 {
40  if (stage != 1) {
41  rbvtrTraCIScenarioManager::initialize(stage);
42  return;
43  }
44  launchConfig = par("launchConfig").xmlValue();
45  seed = par("seed");
46  cXMLElementList basedir_nodes = launchConfig->getElementsByTagName("basedir");
47  if (basedir_nodes.size() == 0) {
48  // default basedir is where current network file was loaded from

50  cXMLElement* basedir_node = new cXMLElement("basedir", __FILE__, launchConfig);

52  launchConfig->appendChild(basedir_node);
53  }
54  cXMLElementList seed_nodes = launchConfig->getElementsByTagName("seed");
55  if (seed_nodes.size() == 0) {
56  if (seed == -1) {
57  // default seed is current repetition
58  const char* seed_s = cSimulation::getActiveSimulation()->getEnvir()->getConfigEx()->getVariable(CFGVAR_RUNNUMBER);
59  seed = atoi(seed_s);
60  }
61  std::stringstream ss; ss << seed;
62  cXMLElement* seed_node = new cXMLElement("seed", __FILE__, launchConfig);
63  seed_node->setAttribute("value", ss.str().c_str());
64  launchConfig->appendChild(seed_node);
65  }
66  rbvtrTraCIScenarioManager::initialize(stage);
67 }

91  rbvtrTraCIBuffer obuf(receiverbvtrTraCIMessage());
92  uint8_t cmdLength; obuf >> cmdLength;

95  std::string description; obuf >> description;
