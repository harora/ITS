94 INetfilter::IHook::Result VanetNetfilterHook::datagramLocalOutHook(IPv4Datagram* datagram, const InterfaceEntry*& outIE, IPv4Address& nextHopAddr)
95 {
96  if (isReactive)
97  {
98  cPacket * packet = dynamic_cast<cPacket *>(datagram);
99  // Dsr routing, Dsr is a HL protocol and send datagram
100  if (datagram->getTransportProtocol()==IP_PROT_DSR)
101  {
102  IPv4ControlInfo *controlInfo = check_and_cast<IPv4ControlInfo *>(packet->getControlInfo());
103  DSRPkt *dsrpkt = check_and_cast<DSRPkt *>(packet);
104  outIE = ift->getInterfaceById(controlInfo->getInterfaceId());
105  nextHopAddr = dsrpkt->nextAddress();
106  }
107 
108  sendRouteUpdateMessageToVanet(datagram);
109 
110  if (checkPacketUnroutable(datagram, outIE))
111  {
112  delete packet->removeControlInfo();
113  sendNoRouteMessageToVanet(datagram);
114  return INetfilter::IHook::STOLEN;
115  }
116  }
117  return INetfilter::IHook::ACCEPT;

122 void VanetNetfilterHook::sendRouteUpdateMessageToVanet(IPv4Datagram *datagram)
123 {
124  if (datagram->getTransportProtocol() != IP_PROT_DSR) // Dsr don't use update code, the Dsr datagram is the update.
125  {
126  ControlVanetRouting *control = new ControlVanetRouting();
127  control->setOptionCode(VANET_ROUTE_UPDATE);
128  control->setSrcAddress(VanetAddress(datagram->getSrcAddress()));
129  control->setDestAddress(VanetAddress(datagram->getDestAddress()));
130  sendToVanet(control);
131  }
132 }

134 void VanetNetfilterHook::sendNoRouteMessageToVanet(IPv4Datagram *datagram)
135 {

145  control->setDestAddress(VanetAddress(datagram->getDestAddress()));
146  control->encapsulate(dynamic_cast<cPacket *>(datagram));

156 bool VanetNetfilterHook::checkPacketUnroutable(IPv4Datagram* datagram, const InterfaceEntry* outIE)
157 {
158  if (outIE != NULL)
159  return false;
160 
161  IPv4Address destAddr = datagram->getDestAddress();
162 
163  if (destAddr.isMulticast() || destAddr.isLimitedBroadcastAddress())
164  return false;
165 
166  if (rt->isLocalAddress(destAddr))
167  return false;
168 
169  return (rt->findBestMatchingRoute(destAddr) == NULL);
170 }
