28 void TwoRayInterferenceModel::filterSignal(AirFrame *frame, const Coord& senderPos, const Coord& receiverPos) {
29  Signal& s = frame->getSignal();
30 
31  const Coord senderPos2D(senderPos.x, senderPos.y);
32  const Coord receiverPos2D(receiverPos.x, receiverPos.y);
33 
34  ASSERT(senderPos.z > 0); // make sure send antenna is above ground
35  ASSERT(receiverPos.z > 0); // make sure receive antenna is above ground
36 
37  double d = senderPos2D.distance(receiverPos2D);
38  double ht = senderPos.z, hr = receiverPos.z;
39 
40  debugEV << "(ht, hr) = (" << ht << ", " << hr << ")" << endl;
41 
42  double d_dir = sqrt( pow (d,2) + pow((ht - hr),2) ); // direct distance
43  double d_ref = sqrt( pow (d,2) + pow((ht + hr),2) ); // distance via ground reflection
44  double sin_theta = (ht + hr)/d_ref;
45  double cos_theta = d/d_ref;
46 
47  double gamma = (sin_theta - sqrt(epsilon_r - pow(cos_theta,2)))/
48  (sin_theta + sqrt(epsilon_r - pow(cos_theta,2)));
49 
50  //is the signal defined to attenuate over frequency?
51  bool hasFrequency = s.getTransmissionPower()->getDimensionSet().hasDimension(Dimension::frequency);
52  debugEV << "Signal contains frequency dimension: " << (hasFrequency ? "yes" : "no") << endl;
53 
54  assert(hasFrequency);
55 
56  debugEV << "Add TwoRayInterferenceModel attenuation (gamma, d, d_dir, d_ref) = (" << gamma << ", " << d << ", " << d_dir << ", " << d_ref << ")" << endl;
57 
58  s.addAttenuation(new TwoRayInterferenceModel::Mapping(gamma, d, d_dir, d_ref, debug));
59 }

67  double lambda = BaseWorldUtility::speedOfLight / freq;
68  double phi = ( 2*M_PI/lambda * (d_dir - d_ref) );

70  1/(sqrt(
71  (pow((1 + gamma * cos(phi)),2)
72  + pow(gamma,2) * pow(sin(phi),2))
73  ))
74  , 2);
