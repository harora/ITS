48 void BaseMacLayer::initialize(int stage)
49 {
50  BaseLayer::initialize(stage);
51 
52  if(stage==0)
53  {
54  // get handle to phy layer
55  if ((phy = FindModule<MacToPhyInterface*>::findSubModule(getParentModule())) == NULL) {
56  error("Could not find a PHY module.");
57  }
58  headerLength = par("headerLength");
59  phyHeaderLength = phy->getPhyHeaderLength();
60 
61  hasPar("coreDebug") ? coreDebug = par("coreDebug").boolValue() : coreDebug = false;
62  }
63  if (myMacAddr == LAddress::L2NULL) {
64  // see if there is an addressing module available
65  // otherwise use NIC modules id as MAC address
66  AddressingInterface* addrScheme = FindModule<AddressingInterface*>::findSubModule(findHost());
67  if(addrScheme) {
68  myMacAddr = addrScheme->myMacAddr(this);
69  } else {
70  const std::string addressString = par("address").stringValue();
71  if (addressString.empty() || addressString == "auto")
72  myMacAddr = LAddress::L2Type(getParentModule()->getId());
73  else
74  myMacAddr = LAddress::L2Type(addressString.c_str());
75  // use streaming operator for string conversion, this makes it more
76  // independent from the myMacAddr type
77  std::ostringstream oSS; oSS << myMacAddr;
78  par("address").setStringValue(oSS.str());
79  }
80  registerInterface();
81  }
82 }

105 MacPkt* BaseMacLayer::encapsMsg(cPacket *netwPkt)
106 {
107  MacPkt *pkt = new MacPkt(netwPkt->getName(), netwPkt->getKind());
108  pkt->setBitLength(headerLength);
109

151 void BaseMacLayer::handleLowerMsg(cMessage *msg)
152 {
153  MacPkt* mac = static_cast<MacPkt *>(msg);
154  LAddress::L2Type dest = mac->getDestAddr();
155  LAddress::L2Type src = mac->getSrcAddr();
156 
157  //only foward to upper layer if message is for me or broadcast
158  if((dest == myMacAddr) || LAddress::isL2Broadcast(dest)) {
159  coreEV << "message with mac addr " << src
160  << " for me (dest=" << dest
161  << ") -> forward packet to upper layer\n";
162  sendUp(decapsMsg(mac));
163  }
164  else{
165  coreEV << "message with mac addr " << src
166  << " not for me (dest=" << dest
167  << ") -> delete (my MAC="<<myMacAddr<<")\n";
168  delete mac;
169  }
170 }
