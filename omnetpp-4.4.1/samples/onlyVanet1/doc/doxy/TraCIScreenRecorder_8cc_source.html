43 void TraCIScreenRecorder::handleMessage(cMessage *msg) {
44  ASSERT(msg == takeScreenshot);
45 
46  // get dirname
47  const char* dirname = par("dirname").stringValue();
48  if (std::string(dirname) == "") {
49  dirname = cSimulation::getActiveSimulation()->getEnvir()->getConfigEx()->getVariable(CFGVAR_RESULTDIR);
50  }
51 
52  // get absolute path of dirname (the TraCI server might be running in a different directory than OMNeT++)
53  std::string dirname_abs;
54  {
55  char* s = realpath(dirname, 0);
56  if (!s) {
57  perror("cannot open output directory");
58  error("cannot open output directory '%s'", dirname);
59  }
60  dirname_abs = s;
61  free(s);
62  }
63 
64  // get filename template
65  std::string filenameTemplate = par("filenameTemplate").stdstringValue();
66  if (filenameTemplate == "") {
67  const char* myRunID = cSimulation::getActiveSimulation()->getEnvir()->getConfigEx()->getVariable(CFGVAR_RUNID);
68  filenameTemplate = "screenshot-" + std::string(myRunID) + "-@%08.2f.png";
69  #ifdef _WIN32
70  // replace ':' with '-'
71  for (std::string::iterator i = filenameTemplate.begin(); i != filenameTemplate.end(); ++i) if (*i == ':') *i = '-';
72  #endif /* _WIN32 */
73  }
74 
75  // assemble filename
76  std::string filename;
77  {
78  std::string tmpl = dirname_abs + "/" + filenameTemplate;
79  char buf[1024];
80  snprintf(buf, 1024, tmpl.c_str(), simTime().dbl());
81  buf[1023] = 0;
82  filename = buf;
83  }
84 
85  // take screenshot
86  TraCIScenarioManager* manager = TraCIScenarioManagerAccess().get();
87  ASSERT(manager);
88  TraCICommandInterface* traci = manager->getCommandInterface();
89  if (!traci) {
90  error("Cannot create screenshot: TraCI is not connected yet");
91  }
92  TraCICommandInterface::GuiView view = traci->guiView(par("viewName"));
93  view.takeScreenshot(filename);
94 
95  // schedule next screenshot
96  simtime_t stop = par("stop");
97  if ((stop == -1) || (simTime() < stop)) {
98  scheduleAt(simTime() + par("interval"), takeScreenshot);
99  }
100 }
