46 void PhyLayer80211p::initialize(int stage) {
47  if (stage == 0) {
48  //get ccaThreshold before calling BasePhyLayer::initialize() which instantiates the deciders
49  ccaThreshold = pow(10, par("ccaThreshold").doubleValue() / 10);
50  collectCollisionStatistics = par("collectCollisionStatistics").boolValue();

52  BasePhyLayer::initialize(stage);
53  if (stage == 0) {
54  if (par("headerLength").longValue() != PHY_HDR_TOTAL_LENGTH) {

56  }
57  //erase the RadioStateAnalogueModel
58  analogueModels.erase(analogueModels.begin());
59  }

366 void PhyLayer80211p::handleSelfMessage(cMessage* msg) {
367 
368  switch(msg->getKind()) {
369  //transmission overBasePhyLayer::
370  case TX_OVER: {
371  assert(msg == txOverTimer);
372  sendControlMsgToMac(new cMessage("Transmission over", TX_OVER));
373  //check if there is another packet on the chan, and change the chan-state to idle
374  Decider80211p* dec = dynamic_cast<Decider80211p*>(decider);
375  assert(dec);
376  if (dec->cca(simTime(),NULL)) {
377  //chan is idle
378  DBG << "Channel idle after transmit!\n";
379  dec->setChannelIdleStatus(true);
380 
381  }
382  else {
383  DBG << "Channel not yet idle after transmit!\n";
384  }
385  break;
386  }
387  //radio switch over
388  case RADIO_SWITCHING_OVER:
389  assert(msg == radioSwitchingOverTimer);
390  BasePhyLayer::finishRadioSwitching();
391  break;
392 
393  //AirFrame
394  case AIR_FRAME:
395  BasePhyLayer::handleAirFrame(static_cast<AirFrame*>(msg));
396  break;
397 
398  //ChannelSenseRequest
399  case CHANNEL_SENSE_REQUEST:
400  BasePhyLayer::handleChannelSenseRequest(msg);
401  break;
402 
403  default:
404  break;
405  }
406 }
407 AirFrame *PhyLayer80211p::encapsMsg(cPacket *macPkt)
408 {
409  // the cMessage passed must be a MacPacket... but no cast needed here
410  // MacPkt* pkt = static_cast<MacPkt*>(msg);
411 
412  // ...and must always have a ControlInfo attached (contains Signal)
413  cObject* ctrlInfo = macPkt->removeControlInfo();
414  assert(ctrlInfo);
415 
416  // create the new AirFrame
417  AirFrame* frame = new AirFrame11p(macPkt->getName(), AIR_FRAME);
418 
419  // Retrieve the pointer to the Signal-instance from the ControlInfo-instance.
420  // We are now the new owner of this instance.
421  Signal* s = MacToPhyControlInfo::getSignalFromControlInfo(ctrlInfo);
422  // make sure we really obtained a pointer to an instance
423  assert(s);
424 
425  // set the members
426  assert(s->getDuration() > 0);
427  frame->setDuration(s->getDuration());
428  // copy the signal into the AirFrame
429  frame->setSignal(*s);
430  //set priority of AirFrames above the normal priority to ensure
431  //channel consistency (before any thing else happens at a time
432  //point t make sure that the channel has removed every AirFrame
433  //ended at t and added every AirFrame started at t)
434  frame->setSchedulingPriority(airFramePriority);
435  frame->setProtocolId(myProtocolId());
436  frame->setBitLength(headerLength);
437  frame->setId(world->getUniqueAirFrameId());
438  frame->setChannel(radio->getCurrentChannel());
439 
440 
441  // pointer and Signal not needed anymore
442  delete s;
443  s = 0;
444 
445  // delete the Control info
446  delete ctrlInfo;
447  ctrlInfo = 0;
448 
449  frame->encapsulate(macPkt);
450 
451  // --- from here on, the AirFrame is the owner of the MacPacket ---
452  macPkt = 0;
453  coreEV <<"AirFrame encapsulated, length: " << frame->getBitLength() << "\n";
454 
455  return frame;
456 }
