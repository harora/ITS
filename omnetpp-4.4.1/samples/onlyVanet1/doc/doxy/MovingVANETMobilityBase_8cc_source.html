78 void MovingVANETMobilityBase::scheduleUpdate()
79 {
80  cancelEvent(moveTimer);
81  if (!stationary && updateInterval != 0) {
82  // periodic update is needed
83  simtime_t nextUpdate = simTime() + updateInterval;
84  if (nextChange != -1 && nextChange < nextUpdate)
85  // next change happens earlier than next update
86  scheduleAt(nextChange, moveTimer);
87  else
88  // next update happens earlier than next change or there is no change at all
89  scheduleAt(nextUpdate, moveTimer);
90  }
91  else if (nextChange != -1)
92  // no periodic update is needed
93  scheduleAt(nextChange, moveTimer);
94 }
