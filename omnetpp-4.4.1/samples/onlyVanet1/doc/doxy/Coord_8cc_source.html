37 static double dist(double coord1, double coord2, double size) {
38  double difference = fabs(coord1 - coord2);
39  if (difference == 0)
40  // NOTE: event if size is zero
41  return 0;
42  else {
43  assert(size != 0);
44  double dist = FWMath::modulo(difference, size);
45  return std::min(dist, size - dist);
