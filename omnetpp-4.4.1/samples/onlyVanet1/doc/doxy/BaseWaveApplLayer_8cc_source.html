62 WaveShortMessage* BaseWaveApplLayer::prepareWSM(std::string name, int lengthBits, t_channel channel, int priority, int rcvId, int serial) {
63  WaveShortMessage* wsm = new WaveShortMessage(name.c_str());
64  wsm->addBitLength(headerLength);
65  wsm->addBitLength(lengthBits);
66 
67  switch (channel) {
68  case type_SCH: wsm->setChannelNumber(Channels::SCH1); break; //will be rewritten at Mac1609_4 to actual Service Channel. This is just so no controlInfo is needed
69  case type_CCH: wsm->setChannelNumber(Channels::CCH); break;
70  }
71  wsm->setPsid(0);
72  wsm->setPriority(priority);
73  wsm->setWsmVersion(1);
74  wsm->setTimestamp(simTime());
75  wsm->setSenderAddress(myId);
76  wsm->setRecipientAddress(rcvId);
77  wsm->setSenderPos(curPosition);
78  wsm->setSerial(serial);
79 
80  if (name == "beacon") {
81  DBG << "Creating Beacon with Priority " << priority << " at Applayer at " << wsm->getTimestamp() << std::endl;
82  }
83  if (name == "data") {
84  DBG << "Creating Data with Priority " << priority << " at Applayer at " << wsm->getTimestamp() << std::endl;
85  }
86 
87  return wsm;
88 }

119 void BaseWaveApplLayer::handleSelfMsg(cMessage* msg) {
