106 void AODVVANETRouting::handleMessage(cMessage *msg)
107 {
108  if (!isOperational) {
109  if (msg->isSelfMessage())
110  throw cRuntimeError("Model error: self msg '%s' received when isOperational is false", msg->getName());
111 
112  EV_ERROR << "Application is turned off, dropping '" << msg->getName() << "' message\n";
113  delete msg;
114  }
115 
116  if (msg->isSelfMessage()) {
117  if (dynamic_cast<WaitForAODVVANETRREP *>(msg))
118  handleWaitForAODVVANETRREP((WaitForAODVVANETRREP *)msg);
119  else if (msg == helloMsgTimer)
120  sendHelloMessagesIfNeeded();
121  else if (msg == expungeTimer)
122  expungeRoutes();
123  else if (msg == counterTimer) {
124  rreqCount = rerrCount = 0;
125  scheduleAt(simTime() + 1, counterTimer);
126  }
127  else if (msg == rrepAckTimer)
128  handleRREPACKTimer();
129  else if (msg == blacklistTimer)
130  handleBlackListTimer();
131  else
132  throw cRuntimeError("Unknown self message");
133  }
134  else {
135  UDPPacket *udpPacket = dynamic_cast<UDPPacket *>(msg);
136  AODVVANETControlPacket *ctrlPacket = check_and_cast<AODVVANETControlPacket *>(udpPacket->decapsulate());
137  IPv4ControlInfo *udpProtocolCtrlInfo = dynamic_cast<IPv4ControlInfo *>(udpPacket->getControlInfo());
138  ASSERT(udpProtocolCtrlInfo != NULL);
139  IPv4Address sourceAddr = udpProtocolCtrlInfo->getSrcAddr();
140  unsigned int arrivalPacketTTL = udpProtocolCtrlInfo->getTimeToLive();
141 
142  switch (ctrlPacket->getPacketType()) {
143  case RREQ:
144  handleRREQ(check_and_cast<AODVVANETRREQ *>(ctrlPacket), sourceAddr, arrivalPacketTTL);
145  break;
146 
147  case RREP:
148  handleRREP(check_and_cast<AODVVANETRREP *>(ctrlPacket), sourceAddr);
149  break;
150 
151  case RERR:
152  handleRERR(check_and_cast<AODVVANETRERR *>(ctrlPacket), sourceAddr);
153  break;
154 
155  case RREPACK:
156  handleRREPACK(check_and_cast<AODVVANETRREPACK *>(ctrlPacket), sourceAddr);
157  break;
158 
159  default:
160  throw cRuntimeError("AODV Control Packet arrived with undefined packet type: %d", ctrlPacket->getPacketType());
161  }
162  delete udpPacket;
163  }

168  Enter_Method("datagramPreRoutingHook");

178  bool isActive = routeData && routeData->isActive();
179  if (isActive && !route->getGateway().isUnspecified()) {
180  EV_INFO << "Active route found: " << route << endl;
181 
182  // Each time a route is used to forward a data packet, its Active Route
183  // Lifetime field of the source, destination and the next hop on the
184  // path to the destination is updated to be no less than the current
185  // time plus ACTIVE_ROUTE_TIMEOUT.
186 
187  //updateValidRouteLifeTime(destAddr, route->getExpirationTime());
188  //updateValidRouteLifeTime(route->getGateway(), route->getExpirationTime());
189 
190  return ACCEPT;
191  }
192  else if (sourceAddr.isUnspecified() || routingTable->isLocalAddress(sourceAddr)) {
193  bool isInactive = routeData && !routeData->isActive();
194  // A node disseminates a RREQ when it determines that it needs a route

263 void AODVVANETRouting::sendRREQ(AODVVANETRREQ *rreq, const IPv4Address& destAddr, unsigned int timeToLive)
264 {
265  // In an expanding ring search, the originating node initially uses a TTL =
266  // TTL_START in the RREQ packet IP header and sets the timeout for
267  // receiving a RREP to RING_TRAVERSAL_TIME milliseconds.
268  // RING_TRAVERSAL_TIME is calculated as described in section 10. The
269  // TTL_VALUE used in calculating RING_TRAVERSAL_TIME is set equal to the
270  // value of the TTL field in the IP header. If the RREQ times out
271  // without a corresponding RREP, the originator broadcasts the RREQ
272  // again with the TTL incremented by TTL_INCREMENT. This continues
273  // until the TTL set in the RREQ reaches TTL_THRESHOLD, beyond which a
274  // TTL = NET_DIAMETER is used for each attempt.
275 
276  if (rreqCount >= rreqRatelimit) {
277  EV_WARN << "A node should not originate more than RREQ_RATELIMIT RREQ messages per second. Canceling sending RREQ" << endl;
278  delete rreq;
279  return;
280  }
281 
282  std::map<IPv4Address, WaitForAODVVANETRREP *>::iterator rrepTimer = waitForRREPTimers.find(rreq->getDestAddr());
283  WaitForAODVVANETRREP *rrepTimerMsg = NULL;
284  if (rrepTimer != waitForRREPTimers.end()) {
285  rrepTimerMsg = rrepTimer->second;
286  unsigned int lastTTL = rrepTimerMsg->getLastTTL();
287  rrepTimerMsg->setDestAddr(rreq->getDestAddr());
288 
289  // The Hop Count stored in an invalid routing table entry indicates the
290  // last known hop count to that destination in the routing table. When
291  // a new route to the same destination is required at a later time
292  // (e.g., upon route loss), the TTL in the RREQ IP header is initially
293  // set to the Hop Count plus TTL_INCREMENT. Thereafter, following each
294  // timeout the TTL is incremented by TTL_INCREMENT until TTL =
295  // TTL_THRESHOLD is reached. Beyond this TTL = NET_DIAMETER is used.
296  // Once TTL = NET_DIAMETER, the timeout for waiting for the RREP is set
297  // to NET_TRAVERSAL_TIME, as specified in section 6.3.
298 
299  if (timeToLive != 0) {
300  rrepTimerMsg->setLastTTL(timeToLive);
301  rrepTimerMsg->setFromInvalidEntry(true);
302  cancelEvent(rrepTimerMsg);
303  }
304  else if (lastTTL + ttlIncrement < ttlThreshold) {
305  ASSERT(!rrepTimerMsg->isScheduled());
306  timeToLive = lastTTL + ttlIncrement;
307  rrepTimerMsg->setLastTTL(lastTTL + ttlIncrement);
308  }
309  else {
310  ASSERT(!rrepTimerMsg->isScheduled());
311  timeToLive = netDiameter;
312  rrepTimerMsg->setLastTTL(netDiameter);
313  }
314  }
315  else {
316  rrepTimerMsg = new WaitForAODVVANETRREP();
317  waitForRREPTimers[rreq->getDestAddr()] = rrepTimerMsg;
318  ASSERT(hasOngoingRouteDiscovery(rreq->getDestAddr()));
319 
320  timeToLive = ttlStart;
321  rrepTimerMsg->setLastTTL(ttlStart);
322  rrepTimerMsg->setFromInvalidEntry(false);
323  rrepTimerMsg->setDestAddr(rreq->getDestAddr());
324  }
325 
326  // Each time, the timeout for receiving a RREP is RING_TRAVERSAL_TIME.
327  simtime_t ringTraversalTime = 2.0 * nodeTraversalTime * (timeToLive + timeoutBuffer);
328  scheduleAt(simTime() + ringTraversalTime, rrepTimerMsg);
329 
330  EV_INFO << "Sending a Route Request with target " << rreq->getDestAddr() << " and TTL= " << timeToLive << endl;
331  sendAODVPacket(rreq, destAddr, timeToLive, jitterPar->doubleValue());
332  rreqCount++;
333 }

335 void AODVVANETRouting::sendRREP(AODVVANETRREP *rrep, const IPv4Address& destAddr, unsigned int timeToLive)
336 {
337  EV_INFO << "Sending Route Reply to " << destAddr << endl;
338 
339  // When any node transmits a RREP, the precursor list for the
340  // corresponding destination node is updated by adding to it
341  // the next hop node to which the RREP is forwarded.
342 
343  IPv4Route *destRoute = routingTable->findBestMatchingRoute(destAddr);
344  const IPv4Address& nextHop = destRoute->getGateway();
345  AODVVANETRouteData *destRouteData = check_and_cast<AODVVANETRouteData *>(destRoute->getProtocolData());
346  destRouteData->addPrecursor(nextHop);
347 
348  // The node we received the Route Request for is our neighbor,
349  // it is probably an unidirectional link
350  if (destRoute->getMetric() == 1) {
351  // It is possible that a RREP transmission may fail, especially if the
352  // RREQ transmission triggering the RREP occurs over a unidirectional
353  // link.
354 
355  rrep->setAckRequiredFlag(true);
356 
357  // when a node detects that its transmission of a RREP message has failed,
358  // it remembers the next-hop of the failed RREP in a "blacklist" set.
359 
360  failedNextHop = nextHop;
361 
362  if (rrepAckTimer->isScheduled())
363  cancelEvent(rrepAckTimer);
364 
365  scheduleAt(simTime() + nextHopWait, rrepAckTimer);
366  }
367  sendAODVPacket(rrep, nextHop, timeToLive, 0);
368 }

370 AODVVANETRREQ *AODVVANETRouting::createRREQ(const IPv4Address& destAddr)
371 {
372  AODVVANETRREQ *rreqPacket = new AODVVANETRREQ("AODV-RREQ");
373 
374  rreqPacket->setGratuitousRREPFlag(askGratuitousRREP);
375  IPv4Route *lastKnownRoute = routingTable->findBestMatchingRoute(destAddr);
376 
377  rreqPacket->setPacketType(RREQ);
378 
379  // The Originator Sequence Number in the RREQ message is the
380  // node's own sequence number, which is incremented prior to
381  // insertion in a RREQ.
382  sequenceNum++;
383 
384  rreqPacket->setOriginatorSeqNum(sequenceNum);
385 
386  if (lastKnownRoute && lastKnownRoute->getSource() == this) {
387  // The Destination Sequence Number field in the RREQ message is the last
388  // known destination sequence number for this destination and is copied
389  // from the Destination Sequence Number field in the routing table.
390 
391  AODVVANETRouteData *routeData = check_and_cast<AODVVANETRouteData *>(lastKnownRoute->getProtocolData());
392  if (routeData && routeData->hasValidDestNum()) {
393  rreqPacket->setDestSeqNum(routeData->getDestSeqNum());
394  rreqPacket->setUnknownSeqNumFlag(false);
395  }
396  else
397  rreqPacket->setUnknownSeqNumFlag(true);
398  }
399  else
400  rreqPacket->setUnknownSeqNumFlag(true); // If no sequence number is known, the unknown sequence number flag MUST be set.
401 
402  rreqPacket->setOriginatorAddr(getSelfIPAddress());
403  rreqPacket->setDestAddr(destAddr);
404 
405  // The RREQ ID field is incremented by one from the last RREQ ID used
406  // by the current node. Each node maintains only one RREQ ID.
407  rreqId++;
408  rreqPacket->setRreqId(rreqId);
409 
410  // The Hop Count field is set to zero.
411  rreqPacket->setHopCount(0);
412 
413  //AODVVANET RREQ ITEMS
414  // Set position, speed, acceleration and direction
415  cModule *host = getContainingNode(this);
416  IVANETMobility *mod = check_and_cast<IVANETMobility *>(host->getSubmodule("mobility"));
417 
418  rreqPacket->setPosition(mod->getCurrentPosition());
419  rreqPacket->setSpeed(mod->getCurrentSpeed());
420  rreqPacket->setAcceleration(mod->getCurrentAcceleration());
421  rreqPacket->setDirection(mod->getCurrentAngularPosition()); //In Rad, 0 being east, with -M_PI <= angle < M_PI.
422 
423  // The TWR is initialized to zero
424  rreqPacket->setTwr(0);
425 
426  // The Expiration time is initialized to a large number
427  rreqPacket->setExpirationTime(10000);
428 
429  // Before broadcasting the RREQ, the originating node buffers the RREQ
430  // ID and the Originator IP address (its own address) of the RREQ for
431  // PATH_DISCOVERY_TIME.
432  // In this way, when the node receives the packet again from its neighbors,
433  // it will not reprocess and re-forward the packet.
434 
435  RREQIdentifier rreqIdentifier(getSelfIPAddress(), rreqId);
436  rreqsArrivalTime[rreqIdentifier] = simTime();
437 
438  return rreqPacket;
439 }

441 AODVVANETRREP *AODVVANETRouting::createRREP(AODVVANETRREQ *rreq, IPv4Route *destRoute, IPv4Route *originatorRoute, const IPv4Address& lastHopAddr)
442 {
443  AODVVANETRREP *rrep = new AODVVANETRREP("AODV-RREP");
444  rrep->setPacketType(RREP);
445 
446  // When generating a RREP message, a node copies the Destination IP
447  // IPv4Address and the Originator Sequence Number from the RREQ message into
448  // the corresponding fields in the RREP message.
449 
450  rrep->setDestAddr(rreq->getDestAddr());
451  rrep->setOriginatorSeqNum(rreq->getOriginatorSeqNum());
452 
453  // OriginatorAddr = The IP address of the node which originated the RREQ
454  // for which the route is supplied.
455  rrep->setOriginatorAddr(rreq->getOriginatorAddr());
456 
457  // Processing is slightly different, depending on whether the node is
458  // itself the requested destination (see section 6.6.1), or instead
459  // if it is an intermediate node with an fresh enough route to the destination
460  // (see section 6.6.2).
461 
462  if (rreq->getDestAddr() == getSelfIPAddress()) { // node is itself the requested destination
463  // If the generating node is the destination itself, it MUST increment
464  // its own sequence number by one if the sequence number in the RREQ
465  // packet is equal to that incremented value.
466 
467  if (!rreq->getUnknownSeqNumFlag() && sequenceNum + 1 == rreq->getDestSeqNum())
468  sequenceNum++;
469 
470  // The destination node places its (perhaps newly incremented)
471  // sequence number into the Destination Sequence Number field of
472  // the RREP,
473  rrep->setDestSeqNum(sequenceNum);
474 
475  // and enters the value zero in the Hop Count field
476  // of the RREP.
477  rrep->setHopCount(0);
478 
479  rrep->setExpirationTime(rreq->getExpirationTime());
480  }
481  else { // intermediate node
482  // it copies its known sequence number for the destination into
483  // the Destination Sequence Number field in the RREP message.
484  AODVVANETRouteData *destRouteData = check_and_cast<AODVVANETRouteData *>(destRoute->getProtocolData());
485  AODVVANETRouteData *originatorRouteData = check_and_cast<AODVVANETRouteData *>(originatorRoute->getProtocolData());
486  rrep->setDestSeqNum(destRouteData->getDestSeqNum());
487 
488  // The intermediate node updates the forward route entry by placing the
489  // last hop node (from which it received the RREQ, as indicated by the
490  // source IP address field in the IP header) into the precursor list for
491  // the forward route entry -- i.e., the entry for the Destination IP
492  // IPv4Address.
493  destRouteData->addPrecursor(lastHopAddr);
494 
495  // The intermediate node also updates its route table entry
496  // for the node originating the RREQ by placing the next hop towards the
497  // destination in the precursor list for the reverse route entry --
498  // i.e., the entry for the Originator IP IPv4Address field of the RREQ
499  // message data.
500 
501  originatorRouteData->addPrecursor(destRoute->getGateway());
502 
503  // The intermediate node places its distance in hops from the
504  // destination (indicated by the hop count in the routing table)
505  // Hop Count field in the RREP.
506 
507  rrep->setHopCount(destRoute->getMetric());
508 
509  // The Lifetime field of the RREP is calculated by subtracting the
510  // current time from the expiration time in its route table entry.
511 
512  rrep->setExpirationTime(rreq->getExpirationTime());
513  }
514  return rrep;
515 }

552 void AODVVANETRouting::handleRREP(AODVVANETRREP *rrep, const IPv4Address& sourceAddr)
553 {
554  EV_INFO << "AODV Route Reply arrived with source addr: " << sourceAddr << " originator addr: " << rrep->getOriginatorAddr()
555  << " destination addr: " << rrep->getDestAddr() << endl;
556 
557  if (rrep->getOriginatorAddr().isUnspecified()) {
558  EV_INFO << "This Route Reply is a Hello Message" << endl;
559  handleHelloMessage(rrep);
560  delete rrep;
561  return;
562  }
563  // When a node receives a RREP message, it searches (using longest-
564  // prefix matching) for a route to the previous hop.
565 
566  // If needed, a route is created for the previous hop,
567  // but without a valid sequence number (see section 6.2)
568 
569  IPv4Route *previousHopRoute = routingTable->findBestMatchingRoute(sourceAddr);
570 
571  double twr = rrep->getTwr();
572  double expirationTime = rrep->getExpirationTime();
573 
574  if (!previousHopRoute || previousHopRoute->getSource() != this) {
575  // create without valid sequence number
576  previousHopRoute = createRoute(sourceAddr, sourceAddr, 1, false, rrep->getOriginatorSeqNum(), true, twr, expirationTime);
577  }
578  else
579  updateRoutingTable(previousHopRoute, sourceAddr, 1, false, rrep->getOriginatorSeqNum(), true, twr, expirationTime);
580 
581  // Next, the node then increments the hop count value in the RREP by one,
582  // to account for the new hop through the intermediate node
583  unsigned int newHopCount = rrep->getHopCount() + 1;
584  rrep->setHopCount(newHopCount);
585 
586  // Then the forward route for this destination is created if it does not
587  // already exist.
588 
589  IPv4Route *destRoute = routingTable->findBestMatchingRoute(rrep->getDestAddr());
590  AODVVANETRouteData *destRouteData = NULL;
591  unsigned int destSeqNum = rrep->getDestSeqNum();
592 
593  if (destRoute && destRoute->getSource() == this) { // already exists
594  destRouteData = check_and_cast<AODVVANETRouteData *>(destRoute->getProtocolData());
595  // Upon comparison, the existing entry is updated only in the following circumstances:
596 
597  // (i) the sequence number in the routing table is marked as
598  // invalid in route table entry.
599 
600  if (!destRouteData->hasValidDestNum()) {
601  updateRoutingTable(destRoute, sourceAddr, newHopCount, true, destSeqNum, true, twr, expirationTime);
602 
603  // If the route table entry to the destination is created or updated,
604  // then the following actions occur:
605  //
606  // - the route is marked as active,
607  //
608  // - the destination sequence number is marked as valid,
609  //
610  // - the next hop in the route entry is assigned to be the node from
611  // which the RREP is received, which is indicated by the source IP
612  // address field in the IP header,
613  //
614  // - the hop count is set to the value of the New Hop Count,
615  //
616  // - the expiry time is set to the current time plus the value of the
617  // Lifetime in the RREP message,
618  //
619  // - and the destination sequence number is the Destination Sequence
620  // Number in the RREP message.
621  }
622  // (ii) the Destination Sequence Number in the RREP is greater than
623  // the node's copy of the destination sequence number and the
624  // known value is valid, or
625  else if (destSeqNum > destRouteData->getDestSeqNum()) {
626  updateRoutingTable(destRoute, sourceAddr, newHopCount, true, destSeqNum, true, twr, expirationTime);
627  }
628  else {
629  // (iii) the sequence numbers are the same, but the route is
630  // marked as inactive, or
631  if (destSeqNum == destRouteData->getDestSeqNum() && !destRouteData->isActive()) {
632  updateRoutingTable(destRoute, sourceAddr, newHopCount, true, destSeqNum, true, twr, expirationTime);
633  }
634  // (iv) the sequence numbers are the same, and the New Hop Count is
635  // smaller than the hop count in route table entry.
636  else if (destSeqNum == destRouteData->getDestSeqNum() && newHopCount < (unsigned int)destRoute->getMetric()) {
637  updateRoutingTable(destRoute, sourceAddr, newHopCount, true, destSeqNum, true, twr, expirationTime);
638  }
639  }
640  }
641  else { // create forward route for the destination: this path will be used by the originator to send data packets
642  destRoute = createRoute(rrep->getDestAddr(), sourceAddr, newHopCount, true, destSeqNum, true, twr, expirationTime);
643  destRouteData = check_and_cast<AODVVANETRouteData *>(destRoute->getProtocolData());
644  }
645 
646  // If the current node is not the node indicated by the Originator IP
647  // IPv4Address in the RREP message AND a forward route has been created or
648  // updated as described above, the node consults its route table entry
649  // for the originating node to determine the next hop for the RREP
650  // packet, and then forwards the RREP towards the originator using the
651  // information in that route table entry.
652 
653  IPv4Route *originatorRoute = routingTable->findBestMatchingRoute(rrep->getOriginatorAddr());
654  if (getSelfIPAddress() != rrep->getOriginatorAddr()) {
655  // If a node forwards a RREP over a link that is likely to have errors or
656  // be unidirectional, the node SHOULD set the 'A' flag to require that the
657  // recipient of the RREP acknowledge receipt of the RREP by sending a RREP-ACK
658  // message back (see section 6.8).
659 
660  if (originatorRoute && originatorRoute->getSource() == this) {
661  //AODVVANETRouteData *originatorRouteData = check_and_cast<AODVVANETRouteData *>(originatorRoute->getProtocolData());
662 
663  // Also, at each node the (reverse) route used to forward a
664  // RREP has its lifetime changed to be the maximum of (existing-
665  // lifetime, (current time + ACTIVE_ROUTE_TIMEOUT).
666 
667  //simtime_t existingLifeTime = originatorRouteData->getLifeTime();
668  //originatorRouteData->setLifeTime(std::max(simTime() + activeRouteTimeout, existingLifeTime));
669 
670  if (simTime() > rebootTime + deletePeriod || rebootTime == 0) {
671  // If a node forwards a RREP over a link that is likely to have errors
672  // or be unidirectional, the node SHOULD set the 'A' flag to require that
673  // the recipient of the RREP acknowledge receipt of the RREP by sending a
674  // RREP-ACK message back (see section 6.8).
675 
676  if (rrep->getAckRequiredFlag()) {
677  AODVVANETRREPACK *rrepACK = createRREPACK();
678  sendRREPACK(rrepACK, sourceAddr);
679  rrep->setAckRequiredFlag(false);
680  }
681 
682  // When any node transmits a RREP, the precursor list for the
683  // corresponding destination node is updated by adding to it
684  // the next hop node to which the RREP is forwarded.
685 
686  destRouteData->addPrecursor(originatorRoute->getGateway());
687 
688  // Finally, the precursor list for the next hop towards the
689  // destination is updated to contain the next hop towards the
690  // source (originator).
691 
692  IPv4Route *nextHopToDestRoute = routingTable->findBestMatchingRoute(destRoute->getGateway());
693  ASSERT(nextHopToDestRoute);
694  AODVVANETRouteData *nextHopToDestRouteData = check_and_cast<AODVVANETRouteData *>(nextHopToDestRoute->getProtocolData());
695  nextHopToDestRouteData->addPrecursor(originatorRoute->getGateway());
696 
697  AODVVANETRREP *outgoingRREP = rrep->dup();
698  forwardRREP(outgoingRREP, originatorRoute->getGateway(), 100);
699  }
700  }
701  else
702  EV_ERROR << "Reverse route doesn't exist. Dropping the RREP message" << endl;
703  }
704  else {
705  if (hasOngoingRouteDiscovery(rrep->getDestAddr())) {
706  EV_INFO << "The Route Reply has arrived for our Route Request to node " << rrep->getDestAddr() << endl;
707  updateRoutingTable(destRoute, sourceAddr, newHopCount, true, destSeqNum, true, twr, expirationTime);
708  completeRouteDiscovery(rrep->getDestAddr());
709  }
710  }
711 
712  delete rrep;
713 }

768 void AODVVANETRouting::handleRREQ(AODVVANETRREQ *rreq, const IPv4Address& sourceAddr, unsigned int timeToLive)
769 {
770  EV_INFO << "AODV Route Request arrived with source addr: " << sourceAddr << " originator addr: " << rreq->getOriginatorAddr()
771  << " destination addr: " << rreq->getDestAddr() << endl;
772 
773  // A node ignores all RREQs received from any node in its blacklist set.
774  std::map<IPv4Address, simtime_t>::iterator blackListIt = blacklist.find(sourceAddr);
775  if (blackListIt != blacklist.end()) {
776  EV_INFO << "The sender node " << sourceAddr << " is in our blacklist. Ignoring the Route Request" << endl;
777  delete rreq;
778  return;
779  }
780 
781  // When a node receives a RREQ, it first creates or updates a route to the previous hop without a valid sequence number.
782  IPv4Route *previousHopRoute = routingTable->findBestMatchingRoute(sourceAddr);
783 
784  if (!previousHopRoute || previousHopRoute->getSource() != this) {
785  // create without valid sequence number
786  previousHopRoute = createRoute(sourceAddr, sourceAddr, 1, false, rreq->getOriginatorSeqNum(), true, rreq->getTwr(), rreq->getExpirationTime());
787  }
788  else
789  updateRoutingTable(previousHopRoute, sourceAddr, 1, false, rreq->getOriginatorSeqNum(), true, rreq->getTwr(), rreq->getExpirationTime());
790 
791  // then checks to determine whether it has received a RREQ with the same Originator IP address and RREQ ID within at least the last
792  // PATH_DISCOVERY_TIME. If such a RREQ has been received, the node silently discards the newly received RREQ.
793  RREQIdentifier rreqIdentifier(rreq->getOriginatorAddr(), rreq->getRreqId());
794  std::map<RREQIdentifier, simtime_t, RREQIdentifierCompare>::iterator checkRREQArrivalTime = rreqsArrivalTime.find(rreqIdentifier);
795  if (checkRREQArrivalTime != rreqsArrivalTime.end() && simTime() - checkRREQArrivalTime->second <= pathDiscoveryTime) {
796  EV_WARN << "The same packet has arrived within PATH_DISCOVERY_TIME= " << pathDiscoveryTime << ". Discarding it" << endl;
797  delete rreq;
798  return;
799  }
800 
801  // update or create
802  rreqsArrivalTime[rreqIdentifier] = simTime();
803 
804  // First, it first increments the hop count value in the RREQ by one, to account for the new hop through the intermediate node.
805  rreq->setHopCount(rreq->getHopCount() + 1);
806 
807  // Then the node searches for a reverse route to the Originator IP Address, using longest-prefix matching.
808  IPv4Route *reverseRoute = routingTable->findBestMatchingRoute(rreq->getOriginatorAddr());
809 
810  // If need be, the route is created, or updated using the Originator Sequence Number from the
811  // RREQ in its routing table.
812  //
813  // When the reverse route is created or updated, the following actions on
814  // the route are also carried out:
815  //
816  // 1. the Originator Sequence Number from the RREQ is compared to the
817  // corresponding destination sequence number in the route table entry
818  // and copied if greater than the existing value there
819  //
820  // 2. the valid sequence number field is set to true;
821  //
822  // 3. the next hop in the routing table becomes the node from which the
823  // RREQ was received (it is obtained from the source IP address in
824  // the IP header and is often not equal to the Originator IP Address
825  // field in the RREQ message);
826  //
827  // 4. the hop count is copied from the Hop Count in the RREQ message;
828  //
829  // Whenever a RREQ message is received, the Lifetime of the reverse
830  // route entry for the Originator IP address is set to be the maximum of
831  // (ExistingLifetime, MinimalLifetime), where
832  //
833  // MinimalLifetime = (current time + 2*NET_TRAVERSAL_TIME - 2*HopCount*NODE_TRAVERSAL_TIME).
834 
835  //AODVVANET RULES: When a node receives RREQ Message:
836  //(1) Extracts movement details and uses this information with its own to compute TWR & ExpirationTime
837  //(2) If Expiration Time is less than current, updates expiration time
838  //(3) Find and store link quality between two nodes
839  //(4) Attach new TWR & Expiration time and own movement details two the RREQ message.
840  // If node doesn't have route to destination flood neighbors with new RREQ message.
841 
842  unsigned int hopCount = rreq->getHopCount();
843 
844  //Extracts movement details and uses this information
845  IVANETMobility *mod = check_and_cast<IVANETMobility *>(host->getSubmodule("mobility"));
846 
847  Coord currentPosition = mod->getCurrentPosition();
848  Coord currentSpeed = mod->getCurrentSpeed();
849  Coord currentAcceleration = mod->getCurrentAcceleration();
850  Coord currentDirection = mod->getCurrentAngularPosition(); //In Rad, 0 being east, with -M_PI <= angle < M_PI.
851 
852  Coord lastPosition = rreq->getPosition();
853  Coord lastSpeed = rreq->getSpeed();
854  Coord lastAcceleration = rreq->getAcceleration();
855  Coord lastDirection = rreq->getDirection(); //In Rad, 0 being east, with -M_PI <= angle < M_PI.
856  double twr = rreq->getTwr();
857 
858  // Compute TWR:
859  // TWR = SUM(f_s*|S_(i-1)-S_i|+f_a*|A_(i-1)-A_i|+f_d*|D_(i-1)-D_i|+[f_q*(1/LQ])
860  // S = speed of the vehicles
861  // f_s = speed weight factor
862  // A = acceleration of the vehicles
863  // f_a = acceleration weight factor
864  // D = direction of the vehicles
865  // f_d = direction weight factor
866  // F_q = Link Quality Weight Factor
867  // LQ = Link Quality between two adjacent vehicles
868 
869  double linkquality = 1;
870  double speedTwr = speedWeight*abs(lastSpeed.x/lastDirection.x - currentSpeed.x/currentSpeed.x);
871  double accelerationTwr = accelerationWeight*abs(lastAcceleration.x/lastDirection.x - currentAcceleration.x/currentSpeed.x);
872  double directionTwr = directionWeight*abs(currentDirection.x-lastDirection.x);
873  double lqTwr = linkQualityWeight/linkquality;
874 
875  twr = twr + speedTwr + accelerationTwr + directionTwr + lqTwr;
876 
877  // Compute ExpirationTime:
878  // expirationTime = (-(a*b+c*d)+sqrt((a*a+c*c)*R*R-(a*d-b*c)*(a*d-b*c)))/(a*a+c*c)
879  // a = vi*(cos(vanglei)-vj(cos(vanglej)
880  // b = xi - xj
881  // c = vi*(sin(vanglei)-vj(sin(vanglej)
882  // d = yi - yj
883  // R is line of site range
884  //virtual Coord getCurrentSpeed() { Coord v = Coord(cos(getAngleRad()), -sin(getAngleRad())); return v * getSpeed(); }
885  //virtual Coord getCurrentAngularPosition() { Coord v = Coord(cos(getAngleRad()), -sin(getAngleRad())); return v; }
886 
887  double a = lastSpeed.x - currentSpeed.x;
888  double b = lastSpeed.x/lastDirection.x - currentSpeed.x/currentSpeed.x;
889  double c = -lastSpeed.y + currentSpeed.y;
890  double d = lastSpeed.y/lastDirection.x - currentSpeed.y/currentSpeed.x;
891 
892  double newExpirationTime = (-(a*b+c*d)+sqrt((a*a+c*c)*losRange*losRange-(a*d-b*c)*(a*d-b*c)))/(a*a+c*c);
893 
894  double expirationTime =std::min(rreq->getExpirationTime(), newExpirationTime);
895 
896  int rreqSeqNum = rreq->getOriginatorSeqNum();
897  if (!reverseRoute || reverseRoute->getSource() != this) { // create
898  // This reverse route will be needed if the node receives a RREP back to the
899  // node that originated the RREQ (identified by the Originator IP Address).
900  reverseRoute = createRoute(rreq->getOriginatorAddr(), sourceAddr, hopCount, true, rreqSeqNum, true, twr, expirationTime);
901  }
902  else {
903  AODVVANETRouteData *routeData = check_and_cast<AODVVANETRouteData *>(reverseRoute->getProtocolData());
904  int routeSeqNum = routeData->getDestSeqNum();
905  int newSeqNum = std::max(routeSeqNum, rreqSeqNum);
906  int newHopCount = rreq->getHopCount(); // Note: already incremented by 1.
907  int routeHopCount = reverseRoute->getMetric();
908  // The route is only updated if the new sequence number is either
909  //
910  // (i) higher than the destination sequence number in the route
911  // table, or
912  //
913  // (ii) the sequence numbers are equal, but the hop count (of the
914  // new information) plus one, is smaller than the existing hop
915  // count in the routing table, or
916  //
917  // (iii) the sequence number is unknown.
918 
919  if (rreqSeqNum > routeSeqNum ||
920  (rreqSeqNum == routeSeqNum && newHopCount < routeHopCount) ||
921  rreq->getUnknownSeqNumFlag())
922  {
923  updateRoutingTable(reverseRoute, sourceAddr, hopCount, true, newSeqNum, true, twr, expirationTime);
924  }
925  }
926 
927  // A node generates a RREP if either:
928  //
929  // (i) it is itself the destination, or
930  //
931  // (ii) it has an active route to the destination, the destination
932  // sequence number in the node's existing route table entry
933  // for the destination is valid and greater than or equal to
934  // the Destination Sequence Number of the RREQ (comparison
935  // using signed 32-bit arithmetic), and the "destination only"
936  // ('D') flag is NOT set.
937 
938  // After a node receives a RREQ and responds with a RREP, it discards
939  // the RREQ. If the RREQ has the 'G' flag set, and the intermediate
940  // node returns a RREP to the originating node, it MUST also unicast a
941  // gratuitous RREP to the destination node.
942 
943  IPv4Route *destRoute = routingTable->findBestMatchingRoute(rreq->getDestAddr());
944  AODVVANETRouteData *destRouteData = destRoute ? dynamic_cast<AODVVANETRouteData *>(destRoute->getProtocolData()) : NULL;
945 
946  // check (i)
947  if (rreq->getDestAddr() == getSelfIPAddress()) {
948  EV_INFO << "I am the destination node for which the route was requested" << endl;
949 
950  // create RREP
951  AODVVANETRREP *rrep = createRREP(rreq, destRoute, reverseRoute, sourceAddr);
952 
953  // send to the originator
954  sendRREP(rrep, rreq->getOriginatorAddr(), 255);
955 
956  delete rreq;
957  return; // discard RREQ, in this case, we do not forward it.
958  }
959 
960  // check (ii)
961  if (destRouteData && destRouteData->isActive() && destRouteData->hasValidDestNum() &&
962  destRouteData->getDestSeqNum() >= rreq->getDestSeqNum())
963  {
964  EV_INFO << "I am an intermediate node who has information about a route to " << rreq->getDestAddr() << endl;
965 
966  if (destRoute->getGateway() == sourceAddr) {
967  EV_WARN << "This RREP would make a loop. Dropping it" << endl;
968 
969  delete rreq;
970  return;
971  }
972 
973  // create RREP
974  AODVVANETRREP *rrep = createRREP(rreq, destRoute, reverseRoute, sourceAddr);
975 
976  // send to the originator
977  sendRREP(rrep, rreq->getOriginatorAddr(), 255);
978 
979  if (rreq->getGratuitousRREPFlag()) {
980  // The gratuitous RREP is then sent to the next hop along the path to
981  // the destination node, just as if the destination node had already
982  // issued a RREQ for the originating node and this RREP was produced in
983  // response to that (fictitious) RREQ.
984 
985  IPv4Route *originatorRoute = routingTable->findBestMatchingRoute(rreq->getOriginatorAddr());
986  AODVVANETRREP *grrep = createGratuitousRREP(rreq, originatorRoute);
987  sendGRREP(grrep, rreq->getDestAddr(), 100);
988  }
989 
990  delete rreq;
991  return; // discard RREQ, in this case, we also do not forward it.
992  }
993  // If a node does not generate a RREP (following the processing rules in
994  // section 6.6), and if the incoming IP header has TTL larger than 1,
995  // the node updates and broadcasts the RREQ to address 255.255.255.255
996  // on each of its configured interfaces (see section 6.14). To update
997  // the RREQ, the TTL or hop limit field in the outgoing IP header is
998  // decreased by one, and the Hop Count field in the RREQ message is
999  // incremented by one, to account for the new hop through the
1000  // intermediate node. (!) Lastly, the Destination Sequence number for the
1001  // requested destination is set to the maximum of the corresponding
1002  // value received in the RREQ message, and the destination sequence
1003  // value currently maintained by the node for the requested destination.
1004  // However, the forwarding node MUST NOT modify its maintained value for
1005  // the destination sequence number, even if the value received in the
1006  // incoming RREQ is larger than the value currently maintained by the
1007  // forwarding node.
1008 
1009  if (timeToLive > 0 && (simTime() > rebootTime + deletePeriod || rebootTime == 0)) {
1010  if (destRouteData)
1011  rreq->setDestSeqNum(std::max(destRouteData->getDestSeqNum(), rreq->getDestSeqNum()));
1012  rreq->setUnknownSeqNumFlag(false);
1013 
1014  AODVVANETRREQ *outgoingRREQ = rreq->dup();
1015  forwardRREQ(outgoingRREQ, timeToLive);
1016  }
1017  else
1018  EV_WARN << "Can't forward the RREQ because of its small (<= 1) TTL: " << timeToLive << " or the AODV reboot has not completed yet" << endl;
1019 
1020  delete rreq;
1021 }
